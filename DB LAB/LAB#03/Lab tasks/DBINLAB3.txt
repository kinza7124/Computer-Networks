1-- Create tables
CREATE TABLE students(
  student_id NUMBER PRIMARY KEY,
  student_name VARCHAR(50) NOT NULL,
  email VARCHAR(50) NOT NULL
);

CREATE TABLE courses(
  course_id NUMBER NOT NULL,
  course_name VARCHAR(50) NOT NULL
);
ALTER TABLE courses ADD CONSTRAINT pk_courseid PRIMARY KEY(course_id);

CREATE TABLE enrollment(
  enrollment_id NUMBER PRIMARY KEY,
  enrollment_date DATE NOT NULL,
  student_id NUMBER NOT NULL,
  course_id NUMBER NOT NULL
);
ALTER TABLE enrollment ADD CONSTRAINT fk_stdid FOREIGN KEY (student_id) REFERENCES students(student_id);
ALTER TABLE enrollment ADD CONSTRAINT fk_crsid FOREIGN KEY (course_id) REFERENCES courses(course_id);
